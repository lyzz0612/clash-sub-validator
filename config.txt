# %y - 2位数年份 (如: 24)

# %m - 月份 (如: 08)
# %d - 日期 (如: 04)
# %H - 小时 (如: 14)
# %M - 分钟 (如: 30)
# %S - 秒 (如: 45)
# %D - 日期YYYYMMDD (如: 20240804)
# %T - 时间HHMMSS (如: 143045)
